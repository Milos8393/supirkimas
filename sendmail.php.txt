<?php
if ($_SERVER["REQUEST_METHOD"] == "POST") {
    // Получаем данные формы
    $carMake = $_POST['carMake'];
    $carModel = $_POST['carModel'];
    $carYear = $_POST['carYear'];
    $carCondition = $_POST['carCondition'];
    $contactName = $_POST['contactName'];
    $contactEmail = $_POST['contactEmail'];

    // Указываем ваш email
    $to = "uabantratu@gmail.com";  // Замените на ваш email
    $subject = "Naujas užklausimas apie automobilio vertinimą";

    // Формируем тело письма
    $message = "
    Naujas užklausimas apie automobilio vertinimą:

    Markė: $carMake
    Modelis: $carModel
    Metai: $carYear
    Būklė: $carCondition

    Vardas: $contactName
    El. paštas: $contactEmail
    ";

    // Заголовки письма
    $headers = "From: no-reply@yourdomain.com" . "\r\n" .
               "Reply-To: $contactEmail" . "\r\n" .
               "X-Mailer: PHP/" . phpversion();

    // Отправляем email
    if (mail($to, $subject, $message, $headers)) {
        echo "Jūsų užklausa sėkmingai išsiųsta!";
    } else {
        echo "Įvyko klaida bandant išsiųsti užklausą. Bandykite dar kartą.";
    }
}
?>
